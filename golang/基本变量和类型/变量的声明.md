>变量的声明以及全局和局部变量

```go
package main
import "fmt"
//全局变量
var n7 = 100
//一次性声明
var (
	n8 = 2.5
	n9 = "nothing"
)

func main(){
	//第一种方式
	var num1 int = 18
	fmt.Println(num1)
	//第二种方式
	var num2 int
	fmt.Println(num2)
	//第三种方式,省略了变量的类型，根据具体的值，来赋予类型
	var num3 = 1
	fmt.Println(num3)
	//第四种方式，省略var，“：=”
	sex := "男"
	fmt.Println(sex)
	fmt.Println("---------------------------")
	//一次声明多个变量
	var n1,n2,n3 int
	fmt.Println(n1)
	fmt.Println(n2)
	fmt.Println(n3)

	var n4,name,n5 = 10,"jack",7.8
	fmt.Println(n4)
	fmt.Println(name)
	fmt.Println(n5)

	n6,height := 2.5,250.25
	fmt.Println(n6)
	fmt.Println(height)

	fmt.Println(n7)
	fmt.Println(n8)
	fmt.Println(n9)
}
```
