两种遍历方式：`for`循环实现遍历,`for range`循环实现遍历
```go
package main

import "fmt"

func main() {
	slice := make([]int, 4, 20)
	slice[0] = 66
	slice[1] = 77
	slice[2] = 88
	slice[3] = 99
	//for循环遍历
	for i := 0; i < len(slice); i++ {
		fmt.Printf("slice[%v] = %v\t", i, slice[i])
	}
	fmt.Println("\n---------------------------------------------")
	//for-range循环遍历
	for i, v := range slice {
		fmt.Printf("slice[%v] = %v\t", i, v)
	}
}
```
