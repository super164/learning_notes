- ### 二维数组的初始化:  
>四种初始化的方式：
```go
package main

import "fmt"

func main() {
	//定义一个二维数组
	var arr1 [2][3]int = [2][3]int{{1, 2, 3}, {4, 5, 6}}
	fmt.Println(arr1)
	//
	var arr2 = [2][3]int{{1, 2, 3}, {4, 5, 6}}
	fmt.Println(arr2)

	var arr3 = [...][]int{{1, 2, 5, 3}, {4, 5, 6}}
	fmt.Println(arr3)

	var arr4 = [...][]int{{2: 1, 0: 2, 1: 3}, {1: 4, 2: 5, 0: 6}}
	fmt.Println(arr4)

}

```

- ### 二维数组的遍历
```go
package main

import "fmt"

func main() {
	var arr = [3][3]int{{1, 4, 7}, {2, 5, 8}, {3, 6, 9}}
	fmt.Println(arr)
	fmt.Println("---------------------")
	//for循环
	for i := 0; i < len(arr); i++ {
		for j := 0; j < len(arr[i]); j++ {
			fmt.Print(arr[i][j], "\t")
		}
		fmt.Println()
	}
	fmt.Println("---------------------")
	//for-range循环
	for i, v1 := range arr {
		for j, v2 := range v1 {
			fmt.Printf("arr[%v][%v]=%v\t", i, j, v2)
		}
		fmt.Println()
	}
}
```
