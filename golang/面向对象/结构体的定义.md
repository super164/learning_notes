```go
package main

import "fmt"

//定义一个结构体
type Teacher struct {
	//变量大写表示外界可以进行访问
	Name   string
	Age    int
	School string
}

func main() {
	//创建结构体实例
	var t1 Teacher
	fmt.Println(t1)
	t1.Name = "牛鬼蛇蛇"
	t1.Age = 11
	t1.School = "牛马大学"
	fmt.Println(t1)
	fmt.Println(t1.Age)
}
```
