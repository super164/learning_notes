# 文件编程实例

统计文件中英文，数字，空格和其他字符的数量

```go
package main

import (
	"bufio"
	"fmt"
	"io"
	"os"
)
type CharCount struct {
	ChCount    int //英文
	NumCount   int //数字
	SpaceCount int //空格
	OtherCount int //其他字符
}
func main() {
	filePath := "e:/test.txt"
	file, err := os.Open(filePath)
	if err != nil {
		fmt.Printf("open file error=%v\n", err)
		return
	}
	defer file.Close()
	var count CharCount
	//创建一个Reader

	reader := bufio.NewReader(file)
	for {
		str, err := reader.ReadString('\n')
		if err == io.EOF {
			break
		}
		for _, v := range str {
			switch {
			case v >= 'a' && v <= 'z':
				count.ChCount++
			case v >= 'A' && v <= 'Z':
				count.ChCount++
			case v == ' ' || v == '\t':
				count.SpaceCount++
			case v >= '0' && v <= '9':
				count.NumCount++
			default:
				count.OtherCount++
			}
		}
	}
	fmt.Printf("字符的个数为：%v,数字的个数为:%v ,空格的个数为：%v,其他字符的个数为：%v", count.ChCount,
		count.NumCount, count.SpaceCount, count.OtherCount)
}

```

创建结构体来统计每个字符的数量
