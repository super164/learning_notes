# 反射快速入门

## 1. 编写一个案例，演示对（基本数据类型，`interface{}`,`reflect.Value`）进行反射的基本操作

```go
package main
import (
	"fmt"
	"reflect"
)
// 专门演示反射
func reflectTest01(b interface{}) {
	//通过反射获取传入的变量的type,kind,值
	rTyp := reflect.TypeOf(b)
	fmt.Printf("rType=%v\n", rTyp)
	//获取reflect.Value
	rVal := reflect.ValueOf(b)

	n2 := 2 + rVal.Int()
	fmt.Println(n2)
	fmt.Printf("rVal=%v rVla type=%T\n", rVal, rVal)

	//将rVal转成interface{}
	iV := rVal.Interface()
	num := iV.(int)
	fmt.Printf("num=%v,num type=%T", num, num)

}
func main() {
	//编写一个案例，演示对（基本数据类型，`interface{}`,`reflect.Value`）进行反射的基本操作
	var num int = 100
	reflectTest01(num)
}
```

反射结构体：

```go
package main
import (
	"fmt"
	"reflect"
)

// 专门演示反射

type Student struct {
	Name string
	Age  int
}
func reflectTest01(b interface{}) {
	//通过反射获取传入的变量的type,kind,值
	rTyp := reflect.TypeOf(b)
	fmt.Printf("rType=%v\n", rTyp)
	//获取reflect.Value
	rVal := reflect.ValueOf(b)

	//将rVal转成interface{}
	iV := rVal.Interface()

	stu1 := iV.(Student)
	fmt.Printf("Name=%v,num type=%T", stu1.Name, stu1)
}
func main() {
	//编写一个案例，演示对（基本数据类型，`interface{}`,`reflect.Value`）进行反射的基本操作
	//定义一个student的一个对象
	stu := Student{
		Name: "tom",
		Age:  18,
	}
	reflectTest01(stu)
}
```
